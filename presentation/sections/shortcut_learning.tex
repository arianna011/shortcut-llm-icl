\section{Shortcut Learning in Large Language Models}

\begin{frame}[fragile]{What is Shortcut Learning?}
\begin{columns}
\begin{column}{0.4\textwidth}
ML models often learn \textbf{non-robust decision rules} (\textcolor{maincolor}{"shortcuts"})\\
\emph{e.g. background $\rightarrow$ object class}\\
\vspace{2em}
\highlight{$\leftarrow$ Plausible Causes}
\begin{itemize}
      \item[\textcolor{sintefdarkgreen}{\textbullet}] simplicity bias
      \item[\textcolor{sintefdarkgreen}{\textbullet}] dataset bias
\end{itemize}
\end{column}

{assets/[cow]}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[fragile]{What is Shortcut Learning?}
\begin{columns}
\begin{column}{0.55\textwidth}
\highlight{$\rightarrow$ Consequences}
\begin{itemize}
      \item[\textcolor{sintefdarkgreen}{\cmark}] Good performance on \textbf{training} examples and ID datasets
      \item[\xmark] Poor generalization on \textbf{OOD data}
      \item[\xmark] Undermined model \textbf{interpretability}
\end{itemize}
\end{column}
